title Upload balance Android

Buyer -> App: upload(amount)
activate Buyer
activate App
App -> WebService: upload(amount, uID)
activate WebService
WebService -> Database: select(uID)
activate Database
Database --> WebService: wID
deactivate Database 
alt wID != error
WebService -> Database: select (wID)
activate Database 
Database --> WebService: balance
deactivate Database 
WebService -> Database: update(wID, balance + amount)
activate Database 
Database --> WebService: message
deactivate Database 
alt message == error
WebService --> App: error
App --> Buyer: error
else
WebService --> App: balance + amount
App --> Buyer: new balance
end alt
else
WebService --> App: error
deactivate WebService
App --> Buyer: error
deactivate App
deactivate Buyer
